Program
{
  Int x = -10;
  
  Int f(Int x, Int y) {
    if(x == y) then {
        return x;
    }
    
    if(x < y) then {
        return f(x + 1, y);
    }
    
    return f(x, y + 1);
    return f(f(x - 1, y * 2), y + 1);
  }
  
  Int inc (Int x ) {
      return x + 1;
  }
  
  Int get_int() extern;
  
  void print_int(Int i) extern;
  
  void main()
  {
      Int y = 0;
      y = get_int();
      if (y < 1) then
        {
            print_int(y);
        }
      else 
        {
            return print_int(inc(y));
        }
  }
}
